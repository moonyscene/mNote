    Animation Properties

    @keyframes{} animation{}

    animation: 모션이름 가속도 동작시간 지연시간 반복횟수 방향지정;

    속성값 순서는 상관없다. 

    @keyframes 모션이름{
	from, 50%, to {
		transform: translate3d(0,0,0);
	}
	25, 75% {
		transform: translate3d(0,0,0);
	}
}
animation-delay: 지연시간
	0s;
	.2s;
	200ms;
animation-direction: 방향지정
	normal;
	reverse;
	alternate;
	alternate-reverse;
animation-duration: 동작시간
	0s;
	.2s;
	200ms;
animation-iteration-count: 반복횟수
	1;	2.3;
	infinite;
animation-name: 모션이름
animation-play-state: 동작상태
	running;
	paused;
animation-timing-function: 가속도
	linear;
	ease;
	ease-in;
	ease-out;
	ease-in-out;
	step-start;
	step-end;
	cubic-bezier(0.250, 0.250, 0.750, 0.750); /* linear */
	cubic-bezier(1.000, 0.000, 0.000, 1.000); /* easeInOutExpo */
animation-fill-mode: 시작전이나 종료후 값 지정.
	none;
	forwards;
	backwards;
	both;
