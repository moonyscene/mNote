    Controlling Animations

    
var tween = gsap.to(".demo4", {
  duration: 4, 
  x: 750, 
  rotation: 360, 
  ease: "none", 
  paused: true
});

// click handlers for controlling the tween instance...
document.querySelector(".demo4 ~ .nav .play").addEventListener('click', function(e){ tween.play(); });
document.querySelector(".demo4 ~ .nav .pause").addEventListener('click', function(e){ tween.pause(); });
document.querySelector(".demo4 ~ .nav .resume").addEventListener('click', function(e){ tween.resume(); });

    Sequencing with Timelines




    
var tl = gsap.timeline({repeat: 30, repeatDelay: 1});

//add 3 tweens that will play in direct succession.
tl.to(".demo5.ex101", {duration: 1, x: 200});
tl.to(".demo5.ex102", {duration: 1, x: 200, scale: 0.2});
tl.to(".demo5.ex103", {duration: 1, x: 200, scale: 2, y: 20});

    Method Chaining




    
var tl = gsap.timeline();

//chain all to() methods together on one line
//we recommend breaking each to() onto its own line for legibility
tl.to(".demo6.ex101", {duration: 1, x: 200})
  .to(".demo6.ex101", {duration: 1, x: 200, scale: 0.2})
  .to(".demo6.ex101", {duration: 1, x: 200, scale: 2, y: 20});

    Timeline Defaults




    
var tl = gsap.timeline({defaults: {duration: 1}});

//no more repitition of duration: 1!
tl.to(".demo7.ex101", {x: 200})
  .to(".demo7.ex102", {duration: 3, x: 200, scale: 0.2}) //inherited default is overwritten
  .to(".demo7.ex103", {x: 200, scale: 2, y: 20});

    Control placement with the Position Parameter

    
tl.to(element, {x: 200})
  .to(element, {y: 200}, "+=1") //1 second after end of timeline (gap)
  .to(element, {rotation:90}, "-=0.5") //0.5 seconds before end of timeline (overlap)
  .to(element, {scale: 4}, 6); //at exactly 6 seconds from the beginning of the timeline (absolute)







    
//create a TimelineMax and make it repeat 3 times with 1 second between iterations
var tl = gsap.timeline({repeat: 3, repeatDelay: 1});

tl.to(".demo8.ex101", {duration: 1, x: 200})
  //start 1 second after previous tween ends (gap)
  .to(".demo8.ex102", {duration: 1, x: 200, scale: 0.2}, "+=1")
  //add a label at the end
  .addLabel("greyAndPink")
  //start both of these animations at the same time, at the "greyAndPink" label.
  .to(".demo8.ex103", {duration: 1, x: 200, scale: 2, y: 20}, "greyAndPink")
  .to(".demo8.ex104", {duration: 1, x: 200, rotation: 360}, "greyAndPink");

    Getter / Setter Methods

    
var tween = gsap.to(".orange", {duration: 1, x: 100});

console.log(tween.duration()); // 1

tween.duration(2); //sets the duration to 2 seconds

var tl = gsap.timeline();
tl.to(".green", {duration: 5, x: 200})
  .to(".orange", {duration: 5, x: 200});
  //the timeline contains 2 sequenced tweens that are both 5 seconds long

console.log(tl.duration()); // 10

//sets the duration to only 2 seconds. For timelines, that actually alters the timeScale to make it play within that duration
tl.duration(2);

    Demo: timeScale()





    
var $restart = document.querySelector(".demo10 ~ .nav .restart"),
	$timeScale02 = document.querySelector(".demo10 ~ .nav .timeScale02"),
	$timeScale1 = document.querySelector(".demo10 ~ .nav .timeScale1"),
	$timeScale4 = document.querySelector(".demo10 ~ .nav .timeScale4"),
	$message = document.querySelector(".demo10 ~ .message");

var tl = gsap.timeline({repeat: 6, yoyo: true, paused: true});

tl.to(".demo10.ex101", {duration: 1, x: 200})
  .to(".demo10.ex102", {duration: 1, x: 200, scale: 0.2})
  .to(".demo10.ex103", {duration: 1, x: 200, scale: 2, y: 20}, "greyAndPink") 
  .to(".demo10.ex104", {duration: 1, x: 200, rotation: 360}, "greyAndPink"); 

$restart.addEventListener('click', function(e){
  tl.restart();
  showTimeScale();
});
$timeScale02.addEventListener('click', function(e){
  tl.timeScale(0.2); //animation will play at 20% normal speed (slower)
  showTimeScale();
});
$timeScale1.addEventListener('click', function(e){
  tl.timeScale(1); //animation will play at normal speed
  showTimeScale();
});
$timeScale4.addEventListener('click', function(e){
  tl.timeScale(4); //animation will play 4 times normal speed (faster)
  showTimeScale();
});

function showTimeScale() {
  $message.innerHTML = "Current timeScale is " +  tl.timeScale(); // get the current timeScale
}

showTimeScale();

    Getting Current Values of an Element's Property

    
let w = gsap.getProperty("#id", "width"); //you can use selector text
let bgColor = gsap.getProperty(element, "backgroundColor"); // or a variable reference

// used in combination with this.targets()
gsap.to(".class", {x: 100, onUpdate: function() {
  let elem = this.targets()[0];
  
  console.log(gsap.getProperty(elem, "x");); // logs all values used for the first element tweened to the console
} });
